## Installation

Install with npm

```bash
  npm install @iconsans/react
```

Install with pnpm

```bash
  pnpm add @iconsans/react
```

Install with yarn

```bash
  yarn add @iconsans/react
```

## Usage/Examples

```jsx
import { Play as PlayBold } from "@iconsans/react/bold";
import { Play as PlayLinear } from "@iconsans/react/linear";

export default function App() {
  return (
    <>
      <PlayBold />
      <PlayLinear />
    </>
  );
}
```
